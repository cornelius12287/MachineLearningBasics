def minMaxScaledAccuracy(n,x_tr,y_tr,x_ts,y_ts):
    knn = KNeighborsClassifier(n)
    knn.fit(x_tr,y_tr)
    predicted = knn.predict(x_ts)
    return np.sum(predicted==y_ts)/len(y_ts)
